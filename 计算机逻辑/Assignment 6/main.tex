\noindent Check whether the following model satisfies the CTL formulas:
\begin{figure}[H]
    \includegraphics[width=0.5\textwidth]{1.png}
\end{figure}
1) $AG(Start \to EF\ Heat)$
\begin{align*}
 AG(Start \to EF\ Heat)&= AG(\neg Start \lor EF\ Heat)\\
 &=AG(\neg (Start \land \neg EF\ Heat)) \\
 &= \neg EF(Start \land \neg EF\ Heat) \\
    &= \neg E[\top\ U(Start \land \neg EF\ Heat)]
\end{align*}

\begin{align*}
& S(M \models Heat) = \{4, 7\} \\
& S(M \models EF\ Heat) = \{1, 2, 3, 4, 5, 6, 7\} \\
& S(M \models \neg EF\ Heat) = \emptyset \\
& S(M \models (Start \wedge \neg EF\ Heat)) = \emptyset \\
& S(M \models E[\top\ U(Start \wedge \neg EF\ Heat)]) = \emptyset \\
& S(M \models \neg E[\top\ U(Start \wedge \neg EF\ Heat)]) = \{1, 2, 3, 4, 5, 6, 7\} \\
& \Rightarrow M \models AG(Start \to EF\ Heat)
\end{align*}

So the model satisfies the formula.
~\\[1em]

2) $AG(A[\neg Error\ U\ (EF\ Close)])$

\begin{align*}
&AG(A[\neg Error\ U\ (EF\ Close)]) \\
&= AG(\neg(E [\neg (EF\ Close)\ U\ (Error \land \neg (EF\ Close))] \lor EG\ \neg (EF\ Close))) \\
&= AG(\neg(E [\neg (EF\ Close)\ U\ (Error \land \neg (EF\ Close))] \lor \neg AF\ (EF\ Close))) \\
&= \neg EF(E [\neg (EF\ Close)\ U\ (Error \land \neg (EF\ Close))] \lor \neg AF\ (EF\ Close)) \\
&= \neg E[\top\ U(E [\neg (EF\ Close)\ U\ (Error \land \neg (EF\ Close))] \lor \neg AF\ (EF\ Close))]
\end{align*}

\begin{align*}
& S(M \models Close) = \{3, 4, 5, 6, 7\} \\
& S(M \models EF \, Close) = \{1, 2, 3, 4, 5, 6, 7\} \\
& S(M \models \neg EF \, Close) = \emptyset \\
& S(M \models \neg AF(EF \, Close)) = \emptyset \\
& S(M \models Error \wedge \neg EF \, Close) = \emptyset \\
& S(M \models E[\neg EF \, Close \, U \, (Error \wedge \neg EF \, Close)]) = \emptyset \\
& S(M \models (E[\neg EF \, Close \, U \, (Error \wedge \neg EF \, Close)] \vee \neg AF(EF \, Close))) = \emptyset \\
& S(M \models E[\top U (E[\neg EF \, Close \, U \, (Error \wedge \neg EF \, Close)] \vee \neg AF(EF \, Close))]) = \emptyset \\
& S(M \models \neg E[\top U (E[\neg EF \, Close \, U \, (Error \wedge \neg EF \, Close)] \vee \neg AF(EF \, Close))]) = \{1, 2, 3, 4, 5, 6, 7\}
\end{align*}

So the model satisfies the formula.
